\subsection{Detailed Proof Placeholder}
\label{app:proof-coercive-bound}

\begin{proof}[Expanded proof of \cref{lem:coercive-bound}]
From \cref{eq:energy} and the growth assumption on $\Phi$,
\[
\mathcal{E}(x)=\frac12\|x\|^2+\Phi(x)
\ge \frac12\|x\|^2-c_0+c_1\|x\|.
\]
Now let $M\in\mathbb{R}$ and define
\[
\Omega_M:=\setof{x\in V}{\mathcal{E}(x)\le M}.
\]
If $\|x\|\to\infty$, the right-hand side above tends to $\infty$, so eventually $\mathcal{E}(x)>M$. Hence $\Omega_M$ is bounded.
\end{proof}

\subsection{Reusable Lemma Skeleton}
\label{app:lemma-skeleton}

\begin{lemma}[Continuity under Composition]
\label{lem:composition-template}
Let $X,Y,Z$ be topological spaces, with $f:X\to Y$ and $g:Y\to Z$ continuous. Then $g\circ f:X\to Z$ is continuous.
\end{lemma}

\begin{proof}
For any open set $U\subseteq Z$, we have
\[
(g\circ f)^{-1}(U)=f^{-1}(g^{-1}(U)).
\]
Since $g$ and $f$ are continuous, $g^{-1}(U)$ is open in $Y$ and then $f^{-1}(g^{-1}(U))$ is open in $X$.
\end{proof}
